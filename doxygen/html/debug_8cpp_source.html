01376         static char result[256];
01377         
01378         char instu[256];
01379         char prefix[3];
01380         Bit16u seg;
01381 
01382         strcpy(instu,inst);
01383         upcase(instu);
01384 
01385         result[0] = 0;
01386         char* pos = strchr(instu,'[');
01387         if (pos) {
01388                 // Segment prefix ?
01389                 if (*(pos-1)==':') {
01390                         char* segpos = pos-3;
01391                         prefix[0] = tolower(*segpos);
01392                         prefix[1] = tolower(*(segpos+1));
01393                         prefix[2] = 0;
01394                         seg = (Bit16u)GetHexValue(segpos,segpos);
01395                 } else {
01396                         if (strstr(pos,"SP") || strstr(pos,"BP")) {
01397                                 seg = SegValue(ss);
01398                                 strcpy(prefix,"ss");
01399                         } else {
01400                                 seg = SegValue(ds);
01401                                 strcpy(prefix,"ds");
01402                         };
01403                 };
01404 
01405                 pos++;
01406                 Bit32u adr = GetHexValue(pos,pos);
01407                 while (*pos!=']') {
01408                         if (*pos=='+') {
01409                                 pos++;
01410                                 adr += GetHexValue(pos,pos);
01411                         } else if (*pos=='-') {
01412                                 pos++;
01413                                 adr -= GetHexValue(pos,pos); 
01414                         } else 
01415                                 pos++;
01416                 };
01417                 Bit32u address = GetAddress(seg,adr);
01418                 if (!(get_tlb_readhandler(address)->flags & PFLAG_INIT)) {
01419                         static char outmask[] = "%s:[%04X]=%02X";
01420                         
01421                         if (cpu.pmode) outmask[6] = '8';
01422                                 switch (DasmLastOperandSize()) {
01423                                 case 8 : {      Bit8u val = mem_readb(address);
01424                                                         outmask[12] = '2';
01425                                                         sprintf(result,outmask,prefix,adr,val);
01426                                                 }       break;
01427                                 case 16: {      Bit16u val = mem_readw(address);
01428                                                         outmask[12] = '4';
01429                                                         sprintf(result,outmask,prefix,adr,val);
01430                                                 }       break;
01431                                 case 32: {      Bit32u val = mem_readd(address);
01432                                                         outmask[12] = '8';
01433                                                         sprintf(result,outmask,prefix,adr,val);
01434                                                 }       break;
01435                         }
01436                 } else {
01437                         sprintf(result,"[illegal]");
01438                 }
01439                 // Variable found ?
01440                 CDebugVar* var = CDebugVar::FindVar(address);
01441                 if (var) {
01442                         // Replace occurence
01443                         char* pos1 = strchr(inst,'[');
01444                         char* pos2 = strchr(inst,']');
01445                         if (pos1 && pos2) {
01446                                 char temp[256];
01447                                 strcpy(temp,pos2);                              // save end

01450                                 strcat(inst,temp);                              // add end

01465                         strcat(inst,"  ("); strcat(inst,descr); strcat(inst,")");

01470         {
01471                 bool jmp = false;

01479                                                 }
01480                                         }       break;
01481                 case 'C' :      {       if (instu[2] == 'X') {

01485                                                 }
01486                                         }       break;

01489                 case 'G' :      {       if (instu[2] == 'E') {

01493                                                 }
01494                                         }       break;

01496                                                         jmp = (get_ZF()?true:false) || ((get_SF()?true:false)!=(get_OF()?true:false)); // JLE

01499                                                 }
01500                                         }       break;

01502                                         }       break;
01503                 case 'N' :      {       switch (instu[2]) {
01504                                                 case 'B' :      
01505                                                 case 'C' :      {       jmp = get_CF()?false:true;      // JNB / JNC
01506                                                                         }       break;
01507                                                 case 'E' :      {       jmp = get_ZF()?false:true;      // JNE

01509                                                 case 'O' :      {       jmp = get_OF()?false:true;      // JNO

01536                                 if (pos) {

01538                                 } else {
01539                                         strcpy(result,"(up)");
01540                                 }
01541                         }
01542                 } else {
01543                         sprintf(result,"(no jmp)");
01544                 }
01545         }
01546         return result;
